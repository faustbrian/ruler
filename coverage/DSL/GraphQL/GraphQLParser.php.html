28/**

30 *
31 * GraphQLParser transforms GraphQL-style filter queries (in JSON string or PHP array
32 * format) into a structured abstract syntax tree of GraphQLNode instances. The parser
33 * supports logical operators (AND/OR/NOT), comparison operators, list operations, string
34 * operations, null checks, type validations, and nested object notation for accessing
35 * deeply nested fields.

78    /**
79     * Parse a specific operator and its operand into an AST node.
80     *
81     * Creates the appropriate node type based on the operator: ComparisonNode
82     * for eq/ne/gt/gte/lt/lte, ListNode for in/notIn, StringNode for string
83     * operations, NullNode for isNull, and TypeNode for isType. The
84     * containsInsensitive operator is normalized to contains with case
85     * insensitivity flag.
86     *
87     * @param string $field    field name or path being tested
88     * @param string $operator operator name defining the type of comparison or test
89     * @param mixed  $value    operand value for the operator
90     *
91     * @throws InvalidArgumentException when the operator is not recognized
92     *
93     * @return GraphQLNode parsed AST node for the specific operator
94     */
95    private static function parseOperator(string $field, string $operator, mixed $value): GraphQLNode
96    {
97        return match ($operator) {
98            'eq', 'ne', 'gt', 'gte', 'lt', 'lte' => new ComparisonNode($operator, $field, $value),
99            'in', 'notIn' => new ListNode($operator, $field, is_array($value) ? $value : throw new InvalidArgumentException($operator.' operator expects an array')),
100            'contains', 'notContains' => new StringNode($operator, $field, is_string($value) ? $value : throw new InvalidArgumentException($operator.' operator expects a string'), false),
101            'containsInsensitive' => new StringNode('contains', $field, is_string($value) ? $value : throw new InvalidArgumentException('containsInsensitive operator expects a string'), true),
102            'notContainsInsensitive' => new StringNode('notContains', $field, is_string($value) ? $value : throw new InvalidArgumentException('notContainsInsensitive operator expects a string'), true),
103            'startsWith', 'endsWith', 'match' => new StringNode($operator, $field, is_string($value) ? $value : throw new InvalidArgumentException($operator.' operator expects a string'), false),
104            'isNull' => new NullNode($field, is_bool($value) ? $value : throw new InvalidArgumentException('isNull operator expects a boolean')),
105            'isType' => new TypeNode($field, is_string($value) ? $value : throw new InvalidArgumentException('isType operator expects a string')),
106            default => throw new InvalidArgumentException('Unsupported operator: '.$operator),
107        };
108    }

206    /**
207     * Parse a single field condition into an AST node.
208     *
209     * Handles three cases: implicit equality for scalar values, nested object
210     * notation for accessing deep fields (converted to dot notation), and
211     * explicit operators on field values. Multiple operators on the same field
212     * are combined with implicit AND logic.
213     *
214     * @param  string      $field field name or path being tested
215     * @param  mixed       $value field value, operator structure, or nested object for deep field access
216     * @return GraphQLNode parsed AST node representing the field condition
217     */
218    private function parseFieldCondition(string $field, mixed $value): GraphQLNode
219    {
220        // Implicit equality
221        if (!is_array($value)) {
222            return new ComparisonNode('eq', $field, $value);
223        }

225        // Nested object (dot notation alternative)
226        /** @var array<string, mixed> $value */
227        if (!self::hasOperators($value)) {
228            // Flatten: {user: {age: {gte: 18}}} => user.age >= 18
229            return $this->parseNestedObject($field, $value);

236            $conditions[] = self::parseOperator($field, $operator, $operandValue);

244    /**

246     *
247     * Converts nested object structures like {user: {age: {gte: 18}}} into
248     * flat field paths like "user.age" for simpler field resolution during
249     * evaluation. Recursively processes nested objects to handle arbitrary
250     * depth. Multiple conditions in a nested object are combined with AND logic.
251     *
252     * @param  string               $prefix accumulated field path prefix from parent levels

272            }
