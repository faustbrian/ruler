43/**
44 * Compiles GraphQL filter AST nodes into executable rule propositions.
45 *
46 * GraphQLCompiler traverses the abstract syntax tree produced by GraphQLParser
47 * and transforms each node into corresponding Proposition instances that can be
48 * evaluated against data contexts. The compiler handles logical combinators,
49 * comparison operations, list operations, string operations, null checks, and
50 * type validations.
51 *
52 * @author Brian Faust <brian@cline.sh>
53 *
54 * @psalm-immutable
55 */

63     *                                     structures using dot notation during evaluation

82     * Compile a single AST node into its corresponding proposition.
83     *
84     * Dispatches to specialized compilation methods based on node type. Supports
85     * logical operations (AND/OR/NOT), comparisons (eq/ne/gt/gte/lt/lte), list
86     * operations (in/notIn), string operations (contains/startsWith/endsWith/match),
87     * null checks, and type validations.
88     *
89     * @param GraphQLNode $node the AST node to compile into a proposition
90     *
91     * @throws RuntimeException when encountering an unknown or unsupported node type
92     *
93     * @return Proposition the compiled proposition for this node
94     */
95    private function compileNode(GraphQLNode $node): Proposition
96    {
97        return match (true) {
98            $node instanceof LogicalNode => $this->compileLogical($node),
99            $node instanceof ComparisonNode => $this->compileComparison($node),
100            $node instanceof ListNode => $this->compileList($node),
101            $node instanceof StringNode => $this->compileString($node),
102            $node instanceof NullNode => $this->compileNull($node),
103            $node instanceof TypeNode => $this->compileType($node),
104            default => throw new RuntimeException('Unknown node type: '.$node::class),
105        };
106    }

110     *
111     * Recursively compiles all child conditions and combines them using the
112     * appropriate logical operator. For NOT operations, the conditions array
113     * should contain exactly one element.
114     *
115     * @param LogicalNode $node the logical operation node containing operator and conditions
116     *
117     * @throws InvalidArgumentException when the operator is not one of: and, or, not
118     *
119     * @return Proposition the compiled logical proposition combining all child conditions
120     */

122    {

192     *
193     * Creates string matching propositions with optional case sensitivity control.
194     * The 'contains' and 'notContains' operators respect the caseInsensitive flag,
195     * while other operators are case-sensitive. The 'match' operator wraps values
196     * in regex delimiters for pattern matching.
197     *
198     * @param StringNode $node the string operation node with field, operator, value, and case sensitivity
199     *
200     * @throws InvalidArgumentException when the operator is not a valid string operator
201     *
202     * @return Proposition the compiled string operation proposition
203     */
204    private function compileString(StringNode $node): Proposition
205    {
206        $field = $this->fieldResolver->resolve($node->field);
207        $value = new Variable(null, $node->value);
208
209        return match ($node->operator) {
210            'contains' => $node->caseInsensitive
211                ? new StringContainsInsensitive($field, $value)
212                : new StringContains($field, $value),
213            'notContains' => $node->caseInsensitive
214                ? new StringDoesNotContainInsensitive($field, $value)
215                : new StringDoesNotContain($field, $value),
216            'startsWith' => new StartsWith($field, $value),
217            'endsWith' => new EndsWith($field, $value),
218            'match' => new Matches($field, new Variable(null, '/'.$node->value.'/')),
219            default => throw new InvalidArgumentException('Unsupported string operator: '.$node->operator),
220        };
221    }
