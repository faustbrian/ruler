19/**
20 * Parser for LDAP filter syntax.
21 *
22 * Implements recursive descent parsing for LDAP filters (RFC 4515).
23 * Converts token stream from LDAPLexer into an Abstract Syntax Tree (AST).
24 *
25 * Grammar:
26 * filter     ::= "(" filtercomp ")"
27 * filtercomp ::= and | or | not | item
28 * and        ::= "&" filterlist
29 * or         ::= "|" filterlist
30 * not        ::= "!" filter
31 * filterlist ::= filter+
32 * item       ::= simple | present | substring
33 * simple     ::= attr filtertype value
34 * filtertype ::= "=" | "~=" | ">=" | "<=" | ">" | "<" | "!="
35 * present    ::= attr "=*"
36 * substring  ::= attr "=" value (contains *)
37 *
38 * @author Brian Faust <brian@cline.sh>
39 */

73     * type, then expects closing parenthesis.
