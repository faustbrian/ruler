27/**
28 * Registry mapping LDAP operators to Ruler operator classes.
29 *
30 * Provides centralized mapping between LDAP filter operator symbols and their
31 * corresponding Ruler operator implementations. Supports both comparison
32 * operators (=, >=, etc.) and logical operators (and, or, not).
33 *
34 * @author Brian Faust <brian@cline.sh>
35 */
36final class LDAPOperatorRegistry
37{
38    /** @var array<string, class-string> Mapping of LDAP comparison operators to Ruler classes */
39    private const array COMPARISON_MAP = [
40        '=' => EqualTo::class,
41        '>=' => GreaterThanOrEqualTo::class,
42        '<=' => LessThanOrEqualTo::class,
43        '>' => GreaterThan::class,
44        '<' => LessThan::class,
45        '!=' => NotEqualTo::class,
46    ];
47
48    /** @var array<string, class-string> Mapping of LDAP logical operators to Ruler classes */
49    private const array LOGICAL_MAP = [
50        'and' => LogicalAnd::class,
51        'or' => LogicalOr::class,
52        'not' => LogicalNot::class,
53    ];
54
55    /**
56     * Retrieves Ruler operator class for LDAP comparison operator.
57     *
58     * @param string $operator LDAP comparison operator symbol: '=', '>=', '<=', '>', '<', or '!='
59     *
60     * @throws InvalidArgumentException If the operator is not recognized
61     *
62     * @return class-string Fully qualified class name of the Ruler comparison operator
63     */
64    public function getComparison(string $operator): string
65    {
66        throw_unless(array_key_exists($operator, self::COMPARISON_MAP), InvalidArgumentException::class, sprintf('Unknown comparison operator: %s', $operator));
67

69    }

71    /**

73     *
74     * @param string $operator LDAP logical operator name: 'and', 'or', or 'not'
75     *
76     * @throws InvalidArgumentException If the operator is not recognized
77     *

80    public function getLogical(string $operator): string
