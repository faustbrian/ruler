55/**
56 * Enhanced Variable with fluent interface for building rule conditions.
57 *
58 * Variables are placeholders in propositions and comparison operators that get
59 * resolved to concrete values during evaluation. This RuleBuilder-enhanced version
60 * extends the base Variable class with a fluent interface for creating variable
61 * properties, operators, and rules without verbose object instantiation.
62 *
63 * The ArrayAccess implementation enables array-style property access:
64 * `$variable['propertyName']` creates or retrieves a VariableProperty.
65 *
66 * Magic method __call() allows dynamic operator registration and invocation,
67 * enabling custom operators to be called as methods on Variable instances.
68 *
69 * @author Brian Faust <brian@cline.sh>
70 *
71 * @implements ArrayAccess<string, mixed>
72 */

103    /**
104     * Dynamically invoke operators registered with the RuleBuilder.
105     *
106     * This magic method enables calling registered operators as methods on the variable.
107     * The operator is looked up in the RuleBuilder's registered namespaces and instantiated
108     * with this variable as the first argument, followed by any provided arguments.
109     *
110     * All non-Variable arguments are automatically wrapped in Variable instances before
111     * being passed to the operator constructor.
112     *
113     * @param string $name the operator method name to invoke (case-sensitive)
114     * @param array<mixed>  $args arguments to pass to the operator constructor
115     *
116     * @throws LogicException when the requested operator is not registered in any namespace
117     *
118     * @return Operator|self returns the created Operator, or a new Variable wrapping
119     *                       a VariableOperand to enable method chaining
120     *
121     * @see RuleBuilder::registerOperatorNamespace
122     * @see RuleBuilder::findOperator
123     */
124    public function __call(string $name, array $args)
125    {
126        /** @var class-string $operatorClass */
127        $operatorClass = $this->ruleBuilder->findOperator($name);
128        $reflection = new ReflectionClass($operatorClass);
129        $args = array_map([$this, 'asVariable'], $args);
130        array_unshift($args, $this);
131
132        $op = $reflection->newInstanceArgs($args);
133
134        if ($op instanceof VariableOperand) {
135            \assert($op instanceof VariableOperator);
136            return $this->wrap($op);
137        }
138
139        \assert($op instanceof Operator);
140        return $op;
141    }
